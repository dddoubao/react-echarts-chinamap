## Install

```bash
npm install --save react-echarts-chinamap
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-echarts-chinamap'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
